<template>
  <div class="insight-card">
    <div class="insight-header">
      <div class="icon-wrapper">
        <slot name="icon"></slot>
      </div>
      <div class="tag">{{ category }}</div>
    </div>
    
    <h3 class="insight-title">{{ title }}</h3>
    <p class="insight-description">{{ description }}</p>
    
    <div class="insight-metrics">
      <div :class="['amount', amountClass]">{{ formattedAmount }}</div>
      <div :class="['change', changeClass]" v-if="showChange">
        {{ changePercentage }} vs last week
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'InsightCard',
  props: {
    title: {
      type: String,
      required: true
    },
    description: {
      type: String,
      required: true
    },
    amount: {
      type: Number,
      required: true
    },
    changePercentage: {
      type: String,
      default: ''
    },
    category: {
      type: String,
      required: true
    },
    type: {
      type: String,
      default: 'neutral',
      validator: (value) => ['positive', 'negative', 'neutral'].includes(value)
    }
  },
  computed: {
    formattedAmount() {
      const prefix = this.amount > 0 ? '+' : '';
      return `${prefix}$${Math.abs(this.amount).toFixed(2)}`;
    },
    amountClass() {
      return this.type;
    },
    changeClass() {
      return this.type;
    },
    showChange() {
      return this.changePercentage !== '';
    }
  }
}
</script>
